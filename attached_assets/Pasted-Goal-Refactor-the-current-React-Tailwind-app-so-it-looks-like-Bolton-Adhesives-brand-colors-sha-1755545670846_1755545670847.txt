4) Buttons – Bolton house style everywhere

Create src/components/ui/Button.tsx and replace all button usages to use this component:

import clsx from "clsx";
type Variant = "primary" | "accent" | "outline" | "ghost";
type Size = "sm"|"md";
export default function Button({variant="primary", size="md", className, ...props}: any){
  const base = "inline-flex items-center justify-center font-medium rounded-xl focus-ring transition active:translate-y-[1px]";
  const sizes = { sm:"h-8 px-3 text-sm", md:"h-10 px-4" };
  const variants: Record<Variant,string> = {
    primary: "bg-primary-600 text-white hover:bg-primary-700 shadow-sm",
    accent:  "bg-accent-500 text-gray-900 hover:brightness-95 shadow-sm",
    outline: "border border-[color:var(--border)] bg-card text-gray-900 hover:border-primary-400",
    ghost:   "bg-transparent text-gray-700 hover:bg-primary-500/5"
  };
  return <button className={clsx(base,sizes[size],variants[variant],className)} {...props} />;
}


Replace the “New Contract” CTA with <Button variant="accent">+ New Contract</Button>; other CTAs use primary.
